Before installing the SDK, ensure you have:

- **An AcountPay account** – You'll need your Client ID from the merchant dashboard
- **A supported environment** – JavaScript/TypeScript (browser or Node.js), React optional

# Installing AcountPay SDK

Choose from multiple installation methods based on your setup.

## Quick Start (Recommended)

For fastest setup, use our CDN:

```html
<script src="https://cdn.acountpay.com/sdk/acount.umd.js"></script>
<script>
  const acount = new Acount({
    clientId: 'your-client-id-here'
  });
</script>
```

## Package Manager Installation

### NPM Installation

```bash
npm i @acountpay/pis-sdk
```

### Yarn Installation

```bash
yarn add @acountpay/pis-sdk
```

## Platform-Specific Integration

### Shopify

1. Go to **Online Store** → **Themes** → **Edit HTML/CSS**
2. Add to `theme.liquid` before `</head>`:

```html
<script src="https://cdn.acountpay.com/sdk/acount.umd.js"></script>
```

3. Add payment button to checkout or product pages:

```html
<button onclick="payWithAcount()">Pay with AcountPay</button>
<script>
function payWithAcount() {
  const acount = new Acount({
    clientId: "{{ settings.acountpay_client_id }}"
  });
  
  acount.initiatePayment({
    amount: {{ cart.total_price | divided_by: 100.0 }},
    referenceNumber: "{{ checkout.order_number }}",
    onSuccess: function(result) {
      window.location.href = "/checkout/success";
    }
  });
}
</script>
```

### WooCommerce

1. Install "Header and Footer Scripts" plugin
2. Add to header scripts:

```html
<script src="https://cdn.acountpay.com/sdk/acount.umd.js"></script>
```

3. Add to checkout page:

```php
<?php
// Add to functions.php
function add_acountpay_button() {
    ?>
    <button id="acountpay-btn">Pay with AcountPay</button>
    <script>
    document.getElementById('acountpay-btn').onclick = function() {
        const acount = new Acount({
            clientId: "<?php echo get_option('acountpay_client_id'); ?>"
        });
        
        acount.initiatePayment({
            amount: <?php echo WC()->cart->get_total('raw'); ?>,
            referenceNumber: "ORDER-<?php echo time(); ?>",
            onSuccess: function(result) {
                window.location.href = "<?php echo wc_get_checkout_url(); ?>?payment=success";
            }
        });
    };
    </script>
    <?php
}
add_action('woocommerce_checkout_after_order_review', 'add_acountpay_button');
?>
```

### Next.js

```typescript
// pages/checkout.tsx
import { useEffect, useState } from 'react';

export default function Checkout() {
  const [acount, setAcount] = useState<any>(null);

  useEffect(() => {
    // Dynamic import for client-side only
    import('@acountpay/pis-sdk').then((module) => {
      const AcountPay = module.default;
      setAcount(new AcountPay({
        clientId: process.env.NEXT_PUBLIC_ACOUNTPAY_CLIENT_ID!
      }));
    });
  }, []);

  const handlePayment = async () => {
    if (acount) {
      await acount.initiateUserPaymentByEmail({
        amount: 99.99,
        requestId: 'ORDER-123',
        callbackURL: `${window.location.origin}/payment-callback`
      });
    }
  };

  return (
    <button onClick={handlePayment} disabled={!acount}>
      Pay with AcountPay
    </button>
  );
}
```

## Package Information

### Contents
- Core payment flow utilities
- TypeScript definitions
- ESM/CJS/UMD builds for broad compatibility
- No embedded UI components (integrates with your existing design)

### Browser Support
- Modern browsers with ES6 support
- Chrome 60+, Firefox 60+, Safari 12+, Edge 79+

### Bundle Sizes
- UMD: ~15KB gzipped
- ESM: ~12KB gzipped
- No external dependencies

## Environment Variables

For framework integrations, set up environment variables:

```bash
# .env file
REACT_APP_ACOUNTPAY_CLIENT_ID=your-client-id
# or
VUE_APP_ACOUNTPAY_CLIENT_ID=your-client-id
# or
NEXT_PUBLIC_ACOUNTPAY_CLIENT_ID=your-client-id
```

## Important Notes

### Global Constructor Name
- **NPM/ESM**: Import as `AcountPay`
- **UMD/Script Tag**: Global constructor is `Acount` (not `AcountPay`)

### Client ID Setup
1. Get your Client ID from AcountPay merchant dashboard
2. Use sandbox Client ID for testing
3. Switch to production Client ID for live payments

### Version Pinning
For production, pin to specific versions:

```html
<!-- Pin to specific version -->
<script src="https://unpkg.com/@acountpay/pis-sdk@1.0.4/dist/acount.umd.js"></script>
```

## 5-Minute Integration Example

Complete working example:

```html
<!DOCTYPE html>
<html>
<head>
  <title>AcountPay Integration</title>
  <script src="https://cdn.acountpay.com/sdk/acount.umd.js"></script>
</head>
<body>
  <h1>Checkout</h1>
  <p>Total: $99.99</p>
  
  <button id="pay-user">Pay with AcountPay Account</button>
  <button id="pay-guest">Pay as Guest</button>

  <script>
    const acount = new Acount({
      clientId: 'your-client-id-here' // Replace with your Client ID
    });

    // User payment
    document.getElementById('pay-user').onclick = async function() {
      try {
        await acount.initiateUserPaymentByEmail({
          amount: 99.99,
          requestId: 'ORDER-' + Date.now(),
          callbackURL: window.location.origin + '/payment-success'
        });
      } catch (error) {
        alert('Payment failed: ' + error.message);
      }
    };

    // Guest payment
    document.getElementById('pay-guest').onclick = function() {
      acount.initiatePayment({
        amount: 99.99,
        referenceNumber: 'ORDER-' + Date.now(),
        onSuccess: function(result) {
          alert('Payment successful!');
          window.location.href = '/order-confirmation';
        },
        onError: function(error) {
          alert('Payment failed: ' + error.message);
        }
      });
    };
  </script>
</body>
</html>
```
